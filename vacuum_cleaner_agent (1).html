How It Works — Agent Logic 
// The agent follows a simple perceive → decide → act loop

function agentStep(rooms, position) {
  const current = rooms[position];

  // PERCEIVE: What is the state of the current room?
  if (current.state === "dirty") {
    // ACT: Clean the room
    current.state = "clean";
    return "cleaned";
  }

  // PERCEIVE: Room is clean — can we move?
  if (position < rooms.length - 1) {
    // ACT: Move to next room
    return "moved";
  }

  // All rooms visited and cleaned → done!
  return "done";
}
